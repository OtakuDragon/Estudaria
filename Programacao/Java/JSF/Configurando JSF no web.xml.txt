Configurando

Após importar todos os jars necessarios para
a utilização do jsf é necessario configurar um
servlet no web.xml que receberá todas as requisições
de paginas .jsf com o codigo abaixo

========================================================
<!-- Cria o servlet que tratará todas as requisições de extensão .jsf --> 
<servlet>
  	<display-name>FacesServlet</display-name>
  	<servlet-name>FacesServlet</servlet-name>
  	<servlet-class>javax.faces.webapp.FacesServlet</servlet-class>
  	<load-on-startup>1</load-on-startup>
</servlet>
<servlet-mapping>
  	<servlet-name>FacesServlet</servlet-name>
	<!--Este valor pode ser .xhtml tambem para que todas as paginas xhtml sejam automaticamente processadas como jsf-->
  	<url-pattern>*.jsf</url-pattern>
</servlet-mapping>


<!--Define em que estagio está o projeto --> 
<!-- o valor Development mostra mensagens de erro mais completas -->
<!--  outros valores possiveis são :Production,UnitTest,SystemTest e Extension --> 
<context-param>
  	<param-name>javax.faces.PROJECT_STAGE</param-name>
  	<param-value>Development</param-value>
</context-param>

<!--Permite que Expressões UEL executem metodos com se fosse linhas de codigo java --> 
<context-param>
  	<param-name>com.sun.faces.expressionFactory</param-name>
  	<param-value>com.sun.el.ExpressionFactoryImpl</param-value>
</context-param>
